\section{What is Git and GitHub Copilot? (Full Explanation)}

\begin{table}[H]

\small
\begin{tabular}{p{3.8cm}p{4.5cm}p{5cm}}
\toprule
\rowcolor{businessgreen!20}
\textbf{Tool} & \textbf{What it is (simple)} & \textbf{Benefit for Mauricio's work as a financial analyst} \\
\midrule
\textbf{Git} & Free version control system (like “track changes on steroids”) & Never lose a version of your LaTeX report or financial model again \\
\textbf{GitHub} & Cloud storage for Git repositories (private and free) & Automatic backup + access from any computer + audit trail \\
\textbf{GitHub Copilot} & AI assistant that writes code and LaTeX for you in real time & Writes 40–60\% of tables, sections, and Python scripts automatically \\
\bottomrule
\end{tabular}
\end{table}

\subsection{Real Example with Your LaTeX Documents}
You write in Spanish:
\begin{verbatim}
% Tabla de KPIs del Macroproyecto Renacimiento Ganadero Maya
\end{verbatim}
→ Press Tab → Copilot instantly generates the complete table with professional formatting.

\subsection{60-Second Setup in VS Code}
\begin{enumerate}
    \item Open your project folder
    \item Ctrl+Shift+P → “Git: Initialize Repository”
    \item Ctrl+Shift+P → “GitHub: Publish to GitHub” (private)
    \item Install “GitHub Copilot” extension and log in
    \item Done! Every save = automatic commit + push
\end{enumerate}
